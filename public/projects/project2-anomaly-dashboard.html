-- Z-Score Anomaly Flag (Power BI DAX)
Anomaly Flag =
VAR _mean = CALCULATE(AVERAGE(Transactions[Amount]), ALLEXCEPT(Transactions, Transactions[Category]))
VAR _std = CALCULATE(STDEV.P(Transactions[Amount]), ALLEXCEPT(Transactions, Transactions[Category]))
VAR _z = DIVIDE(ABS(Transactions[Amount] - _mean), _std, 0)
RETURN IF(_z > [Z-Score Threshold], "ANOMALY", "NORMAL")
