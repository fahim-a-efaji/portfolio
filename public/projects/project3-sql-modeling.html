Window Functions — Running Totals & Rankings
SELECT t.transaction_id, d.full_date, v.vendor_name, t.amount_usd, SUM(t.amount_usd) OVER ( PARTITION BY d.fiscal_year, a.gl_category ORDER BY d.full_date ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW ) AS running_total, RANK() OVER ( PARTITION BY d.fiscal_year ORDER BY t.amount_usd DESC ) AS spend_rank, LAG(t.amount_usd, 1) OVER ( PARTITION BY v.vendor_key ORDER BY d.full_date ) AS prev_transaction, ROUND( (t.amount_usd - AVG(t.amount_usd) OVER(PARTITION BY a.gl_category)) / NULLIF(STDDEV(t.amount_usd) OVER(PARTITION BY a.gl_category), 0), 2 ) AS z_score FROM fact_transactions t JOIN dim_date d ON t.date_key = d.date_key JOIN dim_vendor v ON t.vendor_key = v.vendor_key JOIN dim_account a ON t.account_key = a.account_key WHERE d.fiscal_year = 2025 ORDER BY t.amount_usd DESC LIMIT 10;
Recursive CTE — Approval Chain Traversal
WITH RECURSIVE approval_chain AS ( -- Anchor: start from transactions above threshold SELECT t.transaction_id, t.amount_usd, a.approver_id, a.approver_name, a.approval_level, a.approved_at, 1 AS depth FROM fact_transactions t JOIN approval_steps a ON t.transaction_id = a.transaction_id WHERE a.approval_level = 1 AND t.amount_usd > 100000 UNION ALL -- Recursive: walk up the chain SELECT ac.transaction_id, ac.amount_usd, a.approver_id, a.approver_name, a.approval_level, a.approved_at, ac.depth + 1 FROM approval_chain ac JOIN approval_steps a ON ac.transaction_id = a.transaction_id AND a.approval_level = ac.approval_level + 1 WHERE ac.depth < 5 ) SELECT * FROM approval_chain ORDER BY transaction_id, depth;
Dynamic PIVOT — Spend by Category × Quarter
-- Snowflake dynamic PIVOT SELECT * FROM ( SELECT v.risk_category, 'Q' || d.fiscal_quarter AS quarter, t.amount_usd FROM fact_transactions t JOIN dim_vendor v ON t.vendor_key = v.vendor_key JOIN dim_date d ON t.date_key = d.date_key WHERE d.fiscal_year = 2025 ) PIVOT ( SUM(amount_usd) FOR quarter IN ('Q1', 'Q2', 'Q3', 'Q4') ) AS pivot_table ORDER BY risk_category;
Index Strategy — Query Performance Comparison
-- Clustered index on fact table (Snowflake micro-partition strategy) ALTER TABLE fact_transactions CLUSTER BY (load_ts, account_key, vendor_key); -- Covering index for anomaly queries (SQL Server) CREATE INDEX ix_fact_anomaly ON fact_transactions (is_anomaly, account_key) INCLUDE (transaction_id, amount_usd, z_score, load_ts); -- Partial index — only flag anomalies CREATE INDEX ix_fact_anomaly_partial ON fact_transactions (z_score DESC, account_key) WHERE is_anomaly = 1;
